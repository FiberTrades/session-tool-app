-- ============================================================================ -- Session Tool — Friend requests -- ============================================================================ -- DMs and 1:1 calls become friends-only. A request is sent from the member list, -- lands in the addressee's DM inbox, and must be ACCEPTED before either side can -- message or call. -- -- TWO DELIBERATE EXEMPTIONS -- 1. ADMINS. The coach-mark tour tells members "Help? DM us", and mentors need -- to reach people. So a member can always DM an admin, and an admin can DM -- anyone. Without this the friend gate quietly severs your support channel. -- 2. EXISTING CONVERSATIONS. Anyone who already has a DM thread is grandfathered -- into an accepted friendship, so nobody loses a conversation they are in the -- middle of. -- -- Enforcement is at the DATABASE, not in the UI: the gate is a RESTRICTIVE policy -- on dm_messages, so hiding a button is a courtesy, not the security boundary. -- -- Run once in the Supabase SQL editor. Safe to re-run. -- ============================================================================ -- --------------------------------------------------------------------------- -- 1. The friendship itself -- --------------------------------------------------------------------------- create table if not exists public.friendships ( requester_id uuid not null references auth.users(id) on delete cascade, addressee_id uuid not null references auth.users(id) on delete cascade, status text not null default 'pending' check (status in ('pending','accepted','declined')), created_at timestamptz not null default now(), responded_at timestamptz, primary key (requester_id, addressee_id), check (requester_id <> addressee_id) -- no befriending yourself ); -- One relationship per PAIR, in either direction: this stops A->B and B->A both -- existing as separate pending rows. create unique index if not exists friendships_pair on public.friendships (least(requester_id, addressee_id), greatest(requester_id, addressee_id)); create index if not exists friendships_addressee on public.friendships (addressee_id, status); -- --------------------------------------------------------------------------- -- 2. Are these two allowed to talk? -- --------------------------------------------------------------------------- create or replace function public.st_are_friends(a uuid, b uuid) returns boolean language sql stable security definer set search_path = public as $$ select exists ( select 1 from friendships f where f.status = 'accepted' and ( (f.requester_id = a and f.addressee_id = b) or (f.requester_id = b and f.addressee_id = a) ) ); $$; -- Full members only. Trial and Session Tool members cannot befriend, DM or call. -- -- CAREFUL: "paid" is NOT "full". A Session Tool subscriber is is_paid = true, but they are -- a journal customer, not a community member. This mirrors the app's own _commTier(): -- pl.paid && (plan = 'bundle' | 'mentorship' | 'comp') -> full -- Anything else - trial, sessiontool, null - is 'limited'. Get this wrong and paying -- Session Tool users quietly gain friends, DMs and calls. -- 'comp' is a manual grant, so it counts even if is_paid was never set. create or replace function public.st_is_full(u uuid) returns boolean language sql stable security definer set search_path = public as $$ select exists ( select 1 from profiles p where p.id = u and ( p.plan = 'comp' or (coalesce(p.is_paid, false) and p.plan in ('bundle','mentorship')) ) ); $$; -- Is this user an admin? (reuses whatever st_is_admin() already decides.) create or replace function public.st_user_is_admin(u uuid) returns boolean language sql stable security definer set search_path = public as $$ select exists ( select 1 from profiles p where p.id = u and lower(p.email) = lower('be.o2@hotmail.com') ); $$; -- The single question the DM gate asks. -- 1. the SENDER must be a full member (or an admin) -- 2. and they must be friends (or one of them is an admin) create or replace function public.st_can_dm(a uuid, b uuid) returns boolean language sql stable security definer set search_path = public as $$ select ( st_is_full(a) or st_user_is_admin(a) ) and ( st_are_friends(a, b) or st_user_is_admin(a) -- an admin may reach anyone or st_user_is_admin(b) ); -- and a full member may reach an admin $$; -- --------------------------------------------------------------------------- -- 3. RLS on friendships -- --------------------------------------------------------------------------- alter table public.friendships enable row level security; -- you can see any request you are part of drop policy if exists fr_select on public.friendships; create policy fr_select on public.friendships for select to authenticated using (requester_id = auth.uid() or addressee_id = auth.uid()); -- you may only send a request AS YOURSELF, while not banned, AND as a full member. -- Trial / Session Tool members cannot befriend anyone - so they can never DM or call -- either, since the DM gate requires a friendship. drop policy if exists fr_insert on public.friendships; create policy fr_insert on public.friendships for insert to authenticated with check ( requester_id = auth.uid() and addressee_id <> auth.uid() and status = 'pending' and not st_is_banned() and ( st_is_full(auth.uid()) or st_user_is_admin(auth.uid()) ) ); -- ONLY THE ADDRESSEE may accept or decline. The requester cannot accept their own -- request - which is the whole point of the feature. drop policy if exists fr_update on public.friendships; create policy fr_update on public.friendships for update to authenticated using (addressee_id = auth.uid()) with check (addressee_id = auth.uid() and status in ('accepted','declined')); -- either side may withdraw / unfriend drop policy if exists fr_delete on public.friendships; create policy fr_delete on public.friendships for delete to authenticated using (requester_id = auth.uid() or addressee_id = auth.uid()); -- --------------------------------------------------------------------------- -- 4. Grandfather every existing conversation, BEFORE the gate goes up -- --------------------------------------------------------------------------- insert into public.friendships (requester_id, addressee_id, status, responded_at) select distinct on (least(sender_id, recipient_id), greatest(sender_id, recipient_id)) sender_id, recipient_id, 'accepted', now() from public.dm_messages where sender_id is not null and recipient_id is not null and sender_id <> recipient_id order by least(sender_id, recipient_id), greatest(sender_id, recipient_id), created_at on conflict do nothing; -- --------------------------------------------------------------------------- -- 5. THE GATE. Restrictive, so it is AND-ed with the existing insert policy - -- we do not need to know or touch whatever is already there. -- --------------------------------------------------------------------------- drop policy if exists dm_friends_only on public.dm_messages; create policy dm_friends_only on public.dm_messages as restrictive for insert to authenticated with check ( st_can_dm(sender_id, recipient_id) ); -- --------------------------------------------------------------------------- -- 6. What the client needs, in one call. -- SECURITY DEFINER so it can read the other party's name/avatar from the -- roster without opening profiles up to everyone. -- --------------------------------------------------------------------------- create or replace function public.st_friends() returns table ( other_id uuid, status text, direction text, -- 'in' = they asked you, 'out' = you asked them created_at timestamptz ) language sql stable security definer set search_path = public as $$ select case when f.requester_id = auth.uid() then f.addressee_id else f.requester_id end, f.status, case when f.requester_id = auth.uid() then 'out' else 'in' end, f.created_at from friendships f where (f.requester_id = auth.uid() or f.addressee_id = auth.uid()) and f.status in ('pending','accepted'); $$; -- The client cannot tell who is an admin: st_members() returns role EMOJIS baked into -- the display name, not a flag. Without this the UI would think nobody is an admin and -- block "Help? DM us" - the DB would allow it, but the button would be hidden. create or replace function public.st_admin_ids() returns table (id uuid) language sql stable security definer set search_path = public as $$ select p.id from profiles p where lower(p.email) = lower('be.o2@hotmail.com'); $$; grant execute on function public.st_admin_ids() to authenticated; grant execute on function public.st_is_full(uuid) to authenticated; grant execute on function public.st_friends() to authenticated; grant execute on function public.st_are_friends(uuid, uuid) to authenticated; grant execute on function public.st_can_dm(uuid, uuid) to authenticated; -- --------------------------------------------------------------------------- -- 7. Check -- --------------------------------------------------------------------------- -- Who counts as full? Compare against what the app shows each member. select coalesce(email,'(no email)') as email, plan, is_paid, st_is_full(id) as full_member from public.profiles order by full_member desc, plan nulls last; select count(*) as grandfathered_friendships from public.friendships where status = 'accepted'; select * from public.st_friends();
